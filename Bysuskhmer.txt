local DrRayLibrary = loadstring(game:HttpGet("https://raw.githubusercontent.com/AZYsGithub/DrRay-UI-Library/main/DrRay.lua"))()
local window = DrRayLibrary:Load("Bysuskhmer", "Default")

local tab = DrRayLibrary.newTab("Local Player", "ImageIdHere")

getgenv().loopW = true
getgenv().loopJ = true

tab.newSlider("Speed", "Epic slider", 1000, false, function(num)
getgenv().Walkspeed = num
game:GetService("Players").LocalPlayer.Character.Humanoid.WalkSpeed = Walkspeed
    game:GetService("RunService").Heartbeat:Connect(function()
        if loopW == true then
            pcall(function()
                game:GetService("Players").LocalPlayer.Character.Humanoid.WalkSpeed = Walkspeed
            end)
        end
    end)
end)

tab.newSlider("Jump", "Epic slider", 1000, false, function(num)
getgenv().Jumppower = num
game:GetService("Players").LocalPlayer.Character.Humanoid.JumpPower = Jumppower
    game:GetService("RunService").Heartbeat:Connect(function()
        if loopJ == true then
            pcall(function()
                game:GetService("Players").LocalPlayer.Character.Humanoid.JumpPower = Jumppower
            end)
        end
    end)
end)

tab.newToggle("Infinite Jump", "Toggle! (prints the state)", false, function(toggleState)
    getgenv().InfJ = toggleState
    game:GetService("UserInputService").JumpRequest:connect(function()
        if InfJ == true then
            game:GetService("Players").LocalPlayer.Character:FindFirstChildOfClass'Humanoid':ChangeState("Jumping")
        end
    end)
end)

local game = DrRayLibrary.newTab("Game", "ImageIdHere")

game.newButton("Prison Life", "Prints Hello!", function()
    loadstring(game:HttpGet("https://raw.githubusercontent.com/UhGbaaaa/-/main/Prison%20Life"))()
end)

local esp = DrRayLibrary.newTab("Esp", "ImageIdHere")

esp.newToggle("ESP PLAYER", "Toggle! (prints the state)", false, function(toggleState)
    getgenv().enabled = toggleState --Toggle on/off
getgenv().filluseteamcolor = true --Toggle fill color using player team color on/off
getgenv().outlineuseteamcolor = true --Toggle outline color using player team color on/off
getgenv().fillcolor = Color3.new(0, 0, 0) --Change fill color, no need to edit if using team color
getgenv().outlinecolor = Color3.new(1, 1, 1) --Change outline color, no need to edit if using team color
getgenv().filltrans = 0.5 --Change fill transparency
getgenv().outlinetrans = 0.5 --Change outline transparency

loadstring(game:HttpGet("https://raw.githubusercontent.com/Vcsk/RobloxScripts/main/Highlight-ESP.lua"))()
end)

local sett = DrRayLibrary.newTab("Setting", "ImageIdHere")

sett.newToggle("LoopSpeed", "Toggle! (prints the state)", true, function(toggleState)
    getgenv().loopW = toggleState
end)

sett.newToggle("LoopJump", "Toggle! (prints the state)", true, function(toggleState)
    getgenv().loopJ = toggleState
end)

sett.newButton("Rejoin", "Prints Hello!", function()
    game:GetService("TeleportService"):Teleport(game.PlaceId, game:GetService("Players").LocalPlayer)
end)
